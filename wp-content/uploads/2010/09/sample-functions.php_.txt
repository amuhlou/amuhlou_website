/*
 * Modifying TinyMCE editor to remove unused items.
 */
function customformatTinyMCE($init) {
	// Add block format elements you want to show in dropdown
	$init['theme_advanced_blockformats'] = 'p,h2,h3,h4';
	// Add Special class names, First string is the text that will show up in the menu 
	// After the equals sign is the CSS class name
	// This will make a new "styles" menu appear in tinymce
	$init['theme_advanced_styles'] = 'Left Inset Box=insetBoxLeft,Right Inset Box=insetBoxRight,Lightbox Link=lightBox,Inset Heading Box Left=hBoxLeft,Inset Heading Box Right=hBoxRight,Full-Width Box=fullBox,Video Embed=embed,Video List=videoList';
	
	return $init;
}
// Modify Tiny_MCE init
add_filter('tiny_mce_before_init', 'customformatTinyMCE' );
